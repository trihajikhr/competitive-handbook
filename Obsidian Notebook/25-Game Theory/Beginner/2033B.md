void solve() {
   int n;
   cin >> n;
   vector<int> vec(n);
   for (auto& x : vec) cin >> x;
   ranges::sort(vec);
   cout << vec[n / 2] << '\n';
}

   fastio();
   int t;
   cin >> t;
   while (t--) solve();
