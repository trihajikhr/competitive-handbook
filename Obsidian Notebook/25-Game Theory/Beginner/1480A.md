#include <iostream>
#include <string>
using namespace std;

inline void fastio() {
   ios_base::sync_with_stdio(false);
   cin.tie(nullptr);
   cout.tie(nullptr);
}

void solve() {
   string s;
   cin >> s;
   for (int i = 0; i < s.length(); i++) {
      if (i % 2 == 0) {
         if (s[i] == 'a')
            s[i] = 'b';
         else
            s[i] = 'a';
      } else {
         if (s[i] == 'z')
            s[i] = 'y';
         else
            s[i] = 'z';
      }
   }
   cout << s << '\n';
}

auto main() -> int {
   fastio();
   int t;
   cin >> t;
   while (t--) solve();
   return 0;
}
